function iter(values) {
    let nextIndex = 0;
    // We will return an object
    return {
        next: () => {
            if (nextIndex < values.length) {
                // We will return below object
                return {
                    value: values[nextIndex++],
                    done: false
                }
            } else {
                return { done: true }
            }
        }
    }
}

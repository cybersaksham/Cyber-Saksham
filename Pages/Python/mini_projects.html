"""
Create a library class.
It should contain these functions :-
1. display books
2. Lend book
3. Add book
4. Return book

Create instance of arguments list of books & name of library.
Make dictionary showing lend books & their owner.
Create a main function & run a infinite while loop asking users for their input.
"""

import time


class Library:
    dict_books = {}

    def __init__(self, name, list_books):
        self.name = name
        self.list_books = list_books

    def display(self):
        print("The books available are :-")
        for item in self.list_books:
            print(item)

    def lend(self, lender, lend_book):
        if lend_book in self.list_books:
            if lend_book in self.dict_books:
                print(f"This book is already given to someone else")
            else:
                self.dict_books[lend_book] = lender
                print(f"Book '{lend_book}' successfully lended to {lender}")
        else:
            print("This book is not available in our library.")

    def add(self, add_book):
        if add_book in self.list_books:
            print(f"This book is already in the library")
        else:
            self.list_books.append(add_book)
            print(f"Book '{add_book}' successfully added to library")

    def return_book(self, ret_book):
        if ret_book in self.dict_books:
            del self.dict_books[ret_book]
            print(f"Book '{ret_book}' successfully returned to library")
        elif ret_book in self.list_books:
            print(f"This book is not given to anyone according to our records.")
        else:
            print(f"This book is not available in the library.")

def alarm(set_alarm_timer):
    count = 0
    while True:
        time.sleep(1)
        current_time = datetime.datetime.now()
        now = current_time.strftime("%H:%M:%S")
        date = current_time.strftime("%d/%m/%Y")
        if count == 0:
            print(f"Alarm was set at {now} on {date}")
            print(f"Alarm was set for {set_alarm_timer} on {date}")
            count += 1
        if now == set_alarm_timer:
            print("Time to Wake up")
            print(f"The time is {set_alarm_timer} and date is {date}")
            mixer.music.play()
            break


def actual_time():
    set_alarm_timer = f"{hour.get()}:{min.get()}:{sec.get()}"

import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
from email.mime.base import MIMEBase


def send_mail(email_id, sender_mail, sender_pass, subject, message):
    msg = MIMEMultipart()
    msg['From'] = sender_mail  # Sender's mail id
    msg['To'] = email_id  # Receiver's mail id
    msg['Subject'] = subject  # Subject
    body = message  # Message

    msg.attach(MIMEText(body, 'html'))
    s = smtplib.SMTP('smtp.gmail.com', 587)
    s.starttls()
    s.login(sender_mail, sender_pass)
    text = msg.as_string()
    s.sendmail(sender_mail, email_id, text)
    s.quit()
